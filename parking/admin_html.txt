        function setupSpotConfigurator() {
            const defineNewSpotsBtn = document.getElementById('defineNewSpotsBtn');
            const spotConfigurator = document.getElementById('spotConfigurator');
            const spotsList = document.getElementById('spotsList');
            const cancelConfigBtn = document.getElementById('cancelConfigBtn');
            const saveConfigBtn = document.getElementById('saveConfigBtn');
            const spotCanvas = document.getElementById('spotCanvas');
            const spotList = document.getElementById('spotList');
            const noSpotsMessage = document.getElementById('noSpotsMessage');
            const clearSpotsBtn = document.getElementById('clearSpotsBtn');
            
            // Initialize canvas
            const ctx = spotCanvas.getContext('2d');
            const spots = [];
            let currentSpot = [];
            
            // Show spot configurator
            defineNewSpotsBtn.addEventListener('click', function() {
                spotConfigurator.style.display = 'block';
                spotsList.style.display = 'none';
                
                // Initialize canvas size
                resizeCanvas();
            });
            
            // Hide spot configurator
            cancelConfigBtn.addEventListener('click', function() {
                spotConfigurator.style.display = 'none';
                spotsList.style.display = 'block';
            });
            
            // Save spot configuration
            saveConfigBtn.addEventListener('click', function() {
                // In a real app, this would send the spot data to the API
                alert(`Configuration saved with ${spots.length} parking spots.`);
                
                spotConfigurator.style.display = 'none';
                spotsList.style.display = 'block';
            });
            
            // Clear all spots
            clearSpotsBtn.addEventListener('click', function() {
                if (confirm('Are you sure you want to clear all spots? This cannot be undone.')) {
                    spots.length = 0;
                    currentSpot.length = 0;
                    updateSpotList();
                    drawSpots();
                }
            });
            
            // Canvas click handler
            spotCanvas.addEventListener('click', function(e) {
                const rect = spotCanvas.getBoundingClientRect();
                const x = e.clientX - rect.left;
                const y = e.clientY - rect.top;
                
                // Add point to current spot
                currentSpot.push({x, y});
                
                // If we have 4 points, complete the spot
                if (currentSpot.length === 4) {
                    spots.push([...currentSpot]);
                    currentSpot.length = 0;
                    updateSpotList();
                }
                
                // Redraw canvas
                drawSpots();
            });
            
            // Handle window resize
            window.addEventListener('resize', resizeCanvas);
            
            function resizeCanvas() {
                const videoContainer = document.querySelector('.video-container');
                spotCanvas.width = videoContainer.offsetWidth;
                spotCanvas.height = videoContainer.offsetHeight;
                drawSpots();
            }
            
            function drawSpots() {
                // Clear canvas
                ctx.clearRect(0, 0, spotCanvas.width, spotCanvas.height);
                
                // Draw completed spots
                spots.forEach((spot, index) => {
                    ctx.beginPath();
                    ctx.moveTo(spot[0].x, spot[0].y);
                    
                    for (let i = 1; i < spot.length; i++) {
                        ctx.lineTo(spot[i].x, spot[i].y);
                    }
                    
                    ctx.closePath();
                    ctx.strokeStyle = '#2ecc71';
                    ctx.lineWidth = 2;
                    ctx.stroke();
                    
                    // Fill with semi-transparent color
                    ctx.fillStyle = 'rgba(46, 204, 113, 0.2)';
                    ctx.fill();
                    
                    // Add spot number
                    const centerX = spot.reduce((sum, point) => sum + point.x, 0) / spot.length;
                    const centerY = spot.reduce((sum, point) => sum + point.y, 0) / spot.length;
                    
                    ctx.font = '16px Arial';
                    ctx.fillStyle = '#fff';
                    ctx.textAlign = 'center';
                    ctx.textBaseline = 'middle';
                    ctx.fillText(`${index + 1}`, centerX, centerY);
                });
                
                // Draw current spot being defined
                if (currentSpot.length > 0) {
                    ctx.beginPath();
                    ctx.moveTo(currentSpot[0].x, currentSpot[0].y);
                    
                    for (let i = 1; i < currentSpot.length; i++) {
                        ctx.lineTo(currentSpot[i].x, currentSpot[i].y);
                    }
                    
                    if (currentSpot.length === 3) {
                        // Draw a dashed line back to the first point
                        ctx.setLineDash([5, 5]);
                        ctx.lineTo(currentSpot[0].x, currentSpot[0].y);
                        ctx.setLineDash([]);
                    }
                    
                    ctx.strokeStyle = '#3498db';
                    ctx.lineWidth = 2;
                    ctx.stroke();
                    
                    // Draw points
                    currentSpot.forEach(point => {
                        ctx.beginPath();
                        ctx.arc(point.x, point.y, 5, 0, Math.PI * 2);
                        ctx.fillStyle = '#3498db';
                        ctx.fill();
                    });
                }
            }
            
            function updateSpotList() {
                if (spots.length === 0) {
                    noSpotsMessage.style.display = 'block';
                    spotList.innerHTML = '';
                    return;
                }
                
                noSpotsMessage.style.display = 'none';
                spotList.innerHTML = '';
                
                spots.forEach((spot, index) => {
                    const spotItem = document.createElement('div');
                    spotItem.className = 'spot-item';
                    spotItem.innerHTML = `
                        <div class="spot-name">Spot #${index + 1}</div>
                        <div class="spot-actions">
                            <button class="btn btn-sm btn-outline-danger delete-spot" data-index="${index}">
                                <i class="fas fa-trash-alt"></i>
                            </button>
                        </div>
                    `;
                    
                    spotList.appendChild(spotItem);
                });
                
                // Add event listeners to delete buttons
                document.querySelectorAll('.delete-spot').forEach(button => {
                    button.addEventListener('click', function() {
                        const index = parseInt(this.getAttribute('data-index'));
                        spots.splice(index, 1);
                        updateSpotList();
                        drawSpots();
                    });
                });
            }
        }

            const refreshBtn = document.getElementById('refreshBtn');
